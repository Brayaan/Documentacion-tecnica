41 /// <summary>
42 /// Crea un jugador simulado completo.
43 /// </summary>
44 /// <returns>Componente PlayerAttack del jugador creado.</returns>
45 /// <remarks>
46 /// Este método construye un jugador falso dentro del test:
47 /// <list type="bullet">
48 /// <item><description>GameObject base</description></item>
49 /// <item><description>Hitbox de puño</description></item>

53 /// </list>
54 /// Se usa para aislar la lógica sin depender de la escena real.
55 /// </remarks>
56 private PlayerAttack CreatePlayer()
57 {
58 _playerGO = new GameObject("TestPlayer");
59
60 _punchHitboxGO = new GameObject("PunchHitbox");
61 _punchHitboxGO.transform.SetParent(_playerGO.transform);
62 _punchHitboxGO.SetActive(false);
63
64 _kickHitboxGO = new GameObject("KickHitbox");
65 _kickHitboxGO.transform.SetParent(_playerGO.transform);
66 _kickHitboxGO.SetActive(false);
67
68 _attack = _playerGO.AddComponent<PlayerAttack>();
69 _attack.punchHitbox = _punchHitboxGO;
70 _attack.kickHitbox = _kickHitboxGO;
71
72 _attack.punchCooldown = 0.15f;
73 _attack.kickCooldown = 0.25f;
74
75 return _attack;
76 }

102 [UnityTest]
103 public IEnumerator PunchHitbox_DeactivatesAfterAttackDuration()
104 {
105 CreatePlayer();
106 yield return null;
107
108 // Activar hitbox manualmente
109 _attack.ActivarHitbox();

111
112 // Simular fin de ataque con delay

114
115 yield return new WaitForSeconds(0.4f);
