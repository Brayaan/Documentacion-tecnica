124 /// <summary>
125 /// Verifica que el hitbox solo hace daño cuando el ataque está activo.
126 /// </summary>
127 /// <returns>Coroutine para Unity Test Framework.</returns>
128 /// <remarks>
129 /// <para><b>Casos validados:</b></para>
130 /// <list type="number">
131 /// <item><description>Ataque desactivado → no debe hacer daño</description></item>
132 /// <item><description>Ataque activo → debe aplicar daño</description></item>
133 /// </list>
134 /// </remarks>
135 [UnityTest]
136 public IEnumerator ExtendedHitbox_OnlyDamagesWhenAttackIsActive()
137 {
138 ExtendedHitbox extHitbox = CreateAttacker();
139 EnemyHealthSystem enemyHealth = CreateEnemy();
140
141 yield return null;
142
143 PlayerAttack playerAttack = _attackerGO.GetComponent<PlayerAttack>();
144 Collider2D enemyCol = _targetGO.GetComponent<Collider2D>();
145
146 // Estado: ataque desactivado → no debe hacer daño
147 _isAttackingField.SetValue(playerAttack, false);
148 int initialHealth = enemyHealth.currentHealth;
149
150 _onTriggerEnter2D.Invoke(extHitbox, new object[] { enemyCol });
151
152 Assert.AreEqual(initialHealth, enemyHealth.currentHealth);
153
154 // Estado: ataque activo → debe aplicar daño
155 _isAttackingField.SetValue(playerAttack, true);
156
157 _onTriggerEnter2D.Invoke(extHitbox, new object[] { enemyCol });
158
159 Assert.AreEqual(
160 initialHealth - extHitbox.GetComponent<Hitbox>().damage,
161 enemyHealth.currentHealth
162 );
163 }
