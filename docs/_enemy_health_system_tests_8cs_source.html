42 /// <summary>

55 private EnemyHealthSystem CreateEnemy(Vector3 position = default)
56 {
57 _enemyGO = new GameObject("TestEnemy");
58 _enemyGO.transform.position = position;
59
60 Rigidbody2D rb = _enemyGO.AddComponent<Rigidbody2D>();
